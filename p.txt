procedure fibonnaci(int n, var int res){ 
	var int vFib1 := 1; 
	var int	vFib2 := 0; 
	var int vAuxF := 0; 
	while (vFib2 < n) { 
		vAuxF := vFib1 + vFib2; 
		vFib1 := vFib2; 
		vFib2 := vAuxF; 
	}
	res := vFib2; 
}

	q[0][0][0] + 5;
